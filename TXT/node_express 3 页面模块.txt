//当上面的文件在最终页面模块引入时，是这样的注册方式：
var module01=require('./module01')

$("#sec01").append(module01.html)

var module01process=module01.process($)  

function routerall(app1){

	app1.get('/',module01process,function(req,res){  		
		html=$.html()
		res.send(html)
	})

	module01.ajax(app1)	
}
		
exports.routerall=routerall
